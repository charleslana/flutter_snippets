void main() {
  const int number = 1;
  print(number);

  const double numberDecimal = 1.2;
  print(numberDecimal);

  const num numberBoth = 1;
  print(numberBoth);

  const String string = 'Hello';
  print(string);

  const bool isActive = true;
  print(isActive);
