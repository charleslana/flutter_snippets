// divisão por inteiro é necessário arredondar o número
// utiliza o caractere "~" antes da divisão

void main() {
  int x = 4 ~/ 2;
  int y = 3 ~/ 2;

  print(x);
  print(y);
}

// demais operadores

void main() {
  double a = 4 / 2;
  double b = 3 / 2;
  double c = 1 + 1;
  double d = 2 * 2;
  double e = 1 - 1;
  double f = a % 2;

  var rest = a % 2 == 0 ? 'Par' : 'Impar';

  print(a);
  print(b);
  print(c);
  print(d);
  print(e);
  print(f);
  print(rest);
}
