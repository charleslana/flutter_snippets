// propriedades privada e pública
// para uso de privado usa o caractere "_" antes do campo

class MyClass {
    late int _privateField;
    late int publicField;

    void _privateMethod() => print('privado');
    void publicMethod() => print('público');
}

void main() {
  final myClass = MyClass();

  myClass._privateMethod();
  myClass.publicMethod();
}
